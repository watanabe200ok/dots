```
$ brew install nodebrew
$ nodebrew -v

```

### Node.js setup
```
$ mkdir -p ~/.nodebrew/src
$ nodebrew install-binary latest
$ nodebrew ls
$ nodebrew use v16.4.0
$ nodebrew setup
$ echo 'export PATH=$PATH:$HOME/.nodebrew/current/bin' >> ~/.zshrc
$ source ~/.zshrc
```
